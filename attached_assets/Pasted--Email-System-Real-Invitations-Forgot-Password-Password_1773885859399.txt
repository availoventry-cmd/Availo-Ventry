import nodemailer from "nodemailer";

const transporter = nodemailer.createTransport({
  service: "gmail",
  auth: {
    user: process.env.SMTP_EMAIL,
    pass: process.env.SMTP_PASSWORD,
  },
});

interface SendEmailOptions {
  to: string;
  subject: string;
  html: string;
}

export async function sendEmail({ to, subject, html }: SendEmailOptions): Promise<boolean> {
  try {
    if (!process.env.SMTP_EMAIL || !process.env.SMTP_PASSWORD) {
      console.warn("SMTP not configured — email not sent. Set SMTP_EMAIL and SMTP_PASSWORD in Secrets.");
      return false;
    }
    await transporter.sendMail({
      from: `"Availo Ventry" <${process.env.SMTP_EMAIL}>`,
      to,
      subject,
      html,
    });
    console.log(`Email sent successfully to ${to}`);
    return true;
  } catch (err) {
    console.error("Failed to send email:", err);
    return false;
  }
}

function getBaseUrl(req: any): string {
  const proto = req.headers["x-forwarded-proto"] || req.protocol || "https";
  const host = req.headers["x-forwarded-host"] || req.headers.host;
  return `${proto}://${host}`;
}

export { getBaseUrl };

## PART 3: Backend — Forgot Password + Reset Password Routes

Add these two NEW routes to `artifacts/api-server/src/routes/auth.ts`:

### POST /api/auth/forgot-password (public, no auth required)
```
Request: { email: string }
Logic:
  1. Find user by email (case-insensitive)
  2. If user not found, still return success (don't reveal if email exists)
  3. If user found:
     a. Generate a reset token: generateToken(32)
     b. Set passwordResetToken and passwordResetExpires (30 minutes from now) on the user
     c. Build the reset link: `${getBaseUrl(req)}/reset-password?token=${resetToken}`
     d. Send the email using sendEmail + buildPasswordResetEmail
  4. Return { success: true, message: "If an account exists with that email, a reset link has been sent." }
```

Import at top of auth.ts:
```typescript
import { sendEmail, buildPasswordResetEmail, getBaseUrl } from "../lib/email.js";
import { generateToken } from "../lib/id.js";
```

### POST /api/auth/reset-password (public, no auth required)
```
Request: { token: string, newPassword: string }
Logic:
  1. Find user where passwordResetToken === token AND passwordResetExpires > now
  2. If not found, return 400 "Invalid or expired reset token"
  3. If newPassword.length < 8, return 400
  4. Hash the new password, update the user:
     - passwordHash = hashPassword(newPassword)
     - passwordResetToken = null
     - passwordResetExpires = null
     - passwordChangedAt = new Date()
     - mustChangePassword = false
  5. Return { success: true, message: "Password has been reset. You can now log in." }
```
