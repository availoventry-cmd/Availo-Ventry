# Organization Edit, Detail View & Management Improvements

## Overview
The super admin currently can only create organizations and change their status (activate/suspend/deactivate). There is NO way to edit an organization's details after creation. Add a full edit capability plus a detail view.

## IMPORTANT CONSTRAINTS
- Do NOT delete or recreate existing tables or schemas.
- Do NOT break existing create/status flows.
- The `useUpdateOrganization` hook already exists in the generated API client. Use it.
- Keep all existing functionality working.

### 4c. Resend admin invitation from org dropdown

Add a "Resend Admin Invitation" option in the dropdown menu for orgs that are still in "pending_setup" status. This helps when the first admin didn't receive or lost their invitation email.

In the dropdown menu:
```tsx
{org.status === "pending_setup" && (
  <DropdownMenuItem className="gap-2 cursor-pointer" onClick={() => handleResendAdminInvite(org.id, org.name)}>
    <RefreshCw className="w-4 h-4 text-blue-500" /> Resend Admin Invitation
  </DropdownMenuItem>
)}
```

Create a handler that calls POST `/api/organizations/:orgId/resend-admin-invite`.

Create this endpoint in the backend `organizations.ts`:
```typescript
// POST /api/organizations/:orgId/resend-admin-invite
router.post("/:orgId/resend-admin-invite", requireAuth, requireRole("super_admin"), async (req, res) => {
  try {
    const { orgId } = req.params;
    
    // Find the pending org_admin invitation for this org
    const invitations = await db.select().from(invitationsTable)
      .where(and(
        eq(invitationsTable.orgId, orgId),
        eq(invitationsTable.role, "org_admin"),
        eq(invitationsTable.status, "pending")
      ))
      .limit(1);
    
    if (!invitations[0]) {
      res.status(404).json({ error: "No pending admin invitation found" });
      return;
    }

    const inv = invitations[0];
    const newToken = generateToken();
    
    await db.update(invitationsTable).set({
      invitationToken: newToken,
      tokenExpiresAt: addDays(new Date(), 3),
      resendCount: inv.resendCount + 1,
      lastResentAt: new Date(),
    }).where(eq(invitationsTable.id, inv.id));

    // Send email
    const { sendEmail, buildInvitationEmail, getBaseUrl } = await import("../lib/email.js");
    const [org] = await db.select().from(organizationsTable).where(eq(organizationsTable.id, orgId)).limit(1);
    const baseUrl = getBaseUrl(req);
    const invitationLink = `${baseUrl}/accept-invitation?token=${newToken}`;

    await sendEmail({
      to: inv.email,
      subject: `Reminder: You're invited to manage ${org?.name || "an organization"} on Availo Ventry`,
      html: buildInvitationEmail({
        recipientName: inv.name,
        organizationName: org?.name || "an organization",
        role: "org_admin",
        invitationLink,
        expiresInDays: 3,
      }),
    });

    res.json({ success: true, message: "Admin invitation resent", invitationLink });
  } catch (err) {
    console.error("Resend admin invite error:", err);
    res.status(500).json({ error: "Internal Server Error" });
  }
});
```

For the frontend handler, use a direct fetch call:
```typescript
const handleResendAdminInvite = async (orgId: string, name: string) => {
  try {
    const res = await fetch(`/api/organizations/${orgId}/resend-admin-invite`, { 
      method: "POST", credentials: "include" 
    });
    if (res.ok) {
      toast({ title: "Invitation Resent", description: `Admin invitation for ${name} has been resent.` });
    } else {
      const err = await res.json();
      toast({ title: "Failed", description: err.error || "Could not resend.", variant: "destructive" });
    }
  } catch {
    toast({ title: "Failed", description: "Could not resend invitation.", variant: "destructive" });
  }
};
```
