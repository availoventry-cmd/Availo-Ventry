          organizationName: org?.name || "the organization",

## FIX 3: Shorten QR Pass Code

In `artifacts/api-server/src/lib/id.ts`, change the `generateQrCode` function to produce shorter, more readable codes:

```typescript
export function generateQrCode(): string {
  // Generate 6-character uppercase alphanumeric code (e.g., "A7K2M9")
  const chars = "ABCDEFGHJKLMNPQRSTUVWXYZ23456789"; // excluded I,O,0,1 to avoid confusion
  let code = "";
  const bytes = randomBytes(6);
  for (let i = 0; i < 6; i++) {
    code += chars[bytes[i] % chars.length];
  }
  return code;
}
```

This creates codes like `A7K2M9` instead of `dG9rZW4tZGF0YS1oZXJlLXRlc3Q`. Much easier to read and type.

Also update `generateToken` to produce shorter tracking tokens:

```typescript
export function generateToken(length = 8): string {
  return randomBytes(length).toString("hex");
}
```

NOTE: Existing QR codes in the database will still work — the scan-qr endpoint matches by exact QR code value, so old long codes and new short codes both work.

## FIX 4: Branch Assignment — Allow Org Admin to Edit User's Branch

In `artifacts/ventry/src/pages/portal/settings.tsx`, in the Users tab:

1. Add an "Edit" option to the user dropdown menu (alongside Deactivate/Reactivate)
2. When clicking "Edit", open a dialog that lets the org admin change:
   - User's branch assignment (dropdown of active branches)
   - User's department
   - User's job title

Add state:
```typescript
const [editUserDialog, setEditUserDialog] = useState<any>(null);
const [editUserForm, setEditUserForm] = useState({ branchId: "", department: "", jobTitle: "" });
```

Add handler:
```typescript
const openEditUser = (u: any) => {
  setEditUserDialog(u);
  setEditUserForm({
    branchId: u.branchId || "",
    department: u.department || "",
    jobTitle: u.jobTitle || "",
  });
};

const handleSaveUser = async () => {
  if (!editUserDialog) return;
  try {
    await fetch(`/api/organizations/${user!.orgId}/users/${editUserDialog.id}`, {
      method: "PUT",
      headers: { "Content-Type": "application/json" },
      credentials: "include",
      body: JSON.stringify(editUserForm),
    });
    toast({ title: "User Updated" });
    setEditUserDialog(null);
    queryClient.invalidateQueries({ queryKey: [`/api/organizations/${user?.orgId}/users`] });
  } catch {
    toast({ title: "Failed", variant: "destructive" });
  }
};
```

Add "Edit" to the dropdown menu for each user:
```tsx
<DropdownMenuItem className="cursor-pointer" onClick={() => openEditUser(u)}>
  Edit Details
</DropdownMenuItem>
```

Add edit dialog:

## FIX 5: Improve Email Deliverability (Avoid Junk)

In `artifacts/api-server/src/lib/email.ts`, update the `sendEmail` function to add headers that help avoid spam filters:

```typescript
await transporter.sendMail({
  from: `"Availo Ventry" <${process.env.SMTP_EMAIL}>`,
  to,
  subject,
  html,
  headers: {
    "X-Priority": "1",
    "X-Mailer": "Availo Ventry Platform",
    "List-Unsubscribe": `<mailto:${process.env.SMTP_EMAIL}?subject=unsubscribe>`,
  },
  // Add plain text alternative — emails with both HTML and text are less likely to be marked as spam
  text: html.replace(/<[^>]*>/g, "").replace(/\s+/g, " ").trim(),
});
```

This adds:
- `List-Unsubscribe` header (required by many email providers)
- Plain text alternative (emails without plain text are flagged as spam)
- Priority header
